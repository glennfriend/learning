Object Extend (覆蓋)


    var objA = {"name": "aa", "age": 17};
    var objB = {"name": "bb", "car": "suzuki"};
    var objC = {"name": "cc", "car": undefined};
    console.log(
        _.assign(objA, objB, objC)
    );

Removing properties


    var obj = {
        "name":  "john",
        "age":   17,
        "car":   "suzuki",
        "marry": false
    };
    console.log(
        _.omit(obj, ['car', 'age'])
    );

只取得部份屬性


    var obj = {
        "name":  "john",
        "age":   17,
        "car":   "suzuki",
        "marry": false
    };
    console.log(
        _.pick(obj, ['car', 'age', 'marry'])
    );

Error Handling


    function parseJson(str){
        return _.attempt(JSON.parse.bind(null, str));
    }

    var data = parseJson('a');

    var data = parseJson('{"name": "colin"}');
