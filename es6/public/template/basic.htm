let 宣告區域變數


    let i = 100;
    for ( let i=0 ; i<3; i++ ) {
        console.log(i);
    }
    console.log( '=> ' + i );

    var nums = [1,2,3];

    nums.forEach( function(v) {

    });

    nums.forEach( (v) => {

    });

Destructuring


    var arr = [1,2,3];
    var [a, , b] = arr;
    var {x} = {x:100, y:200};
    console.log('a = ' + a);
    console.log('b = ' + b);
    console.log('x = ' + x);
