請使用 reject 代替 throw


var promise = new Promise(function(resolve, reject){
    reject(new Error("has error"));
});

promise
.catch(function(error){
    console.error(error);
})

不要使用 throw 的方式


var promise = new Promise(function(resolve, reject){
    throw new Error("has error");
});

promise
.catch(function(error){
    console.error(error);
});
