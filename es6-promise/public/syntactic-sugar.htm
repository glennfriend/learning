resolve 基本型

var promise = new Promise(function(resolve, reject){
    resolve(200);
});

promise
.then(function(value){
    console.info(value);
})
.catch(function(error){
    console.warn(error);
});

resolve 語法糖衣

Promise.resolve(200)
.then(function(value){
    console.info(value);
})
.catch(function(error){
    console.warn(error);
});
